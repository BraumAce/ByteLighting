class Solution {
public:
    vector<vector<int>> levelOrder(TreeNode* root) {
        vector<vector<int>> res;
        if (!root) return res;

        queue<TreeNode*> q;
        q.push(root);

        while (!q.empty()) {
            int len = q.size();
            vector<int> cur;
            
            for (int i = 0; i < len; i++) {
                TreeNode* node = q.front();
                q.pop();
                cur.push_back(node->val);
                
                if (node->left) q.push(node->left);
                if (node->right) q.push(node->right);
            }
            
            res.push_back(cur);
        }
        
        return res;
    }
};

class Solution {
public:
    int countWays(int n) {
        const int mod = 1e9 + 7;

        int a[] = {1, 5, 10, 25};
        vector<int> dp(n + 1, 0);
        dp[0] = 1;

        for (int x : a) {
            for (int i = x; i <= n; ++i) {
                dp[i] = (dp[i] + dp[i - x]) % mod;
            }
        }
        
        return dp[n];
    }
};
